\chapter{Instrukcja uruchomieniowa}

Aby poprawnie uruchomić witrynę w celach testowych należy wykonać następujące kroki uprzednio przygotowując środowisko uruchomieniowe:\newline

\textbf{Wymagane środowisko}
\begin{itemize}
    \item \textbf{Proponowany system operacyjny:} Microsoft Windows / Linux / Apple MacOS
    \item \textbf{IDE:} Visual Studio Code
    \item \textbf{Rozszerzenia:} Live Server autorstwa: \textbf{Ritwick'a Dey'a}
    \item \textbf{Repozytorium GitHub:} \url{https://github.com/Avsh11/web-technologies}
    \item \textbf{Proponowany system kontroli wersji:} GitHub Desktop App
\end{itemize}

\textbf{Krok 1. - Konfiguracja środowiska testowego}
\begin{enumerate}
    \item Pobierz edytor \texttt{Visual Studio Code}.
    \item Pobierz aplikację \texttt{GitHub Desktop}.
    \item Uruchom apliakcje \texttt{GitHub Desktop} i wybierz opcje sklonowania repozytorium w zakładce \textbf{File}.
    \item Sklonowane repozytorium umieszczone ma zostać w wybranym stworzonym przez użytkownika katalogu.
\end{enumerate}
\textbf{Krok 2. - Uruchomienie strony}
\begin{enumerate}
    \item Otwórz uprzednio skonfigurowaną aplikację \texttt{Visual Studio Code}.
    \item Upewnij się, że poprawnie zainstalowałeś rozszerzenie \texttt{Live Server}.
    \item Otwórz katalog, w którym sklonowałeś repozytorium.
    \item Wejdź w plik \texttt{index.html}.
    \item W prawym dolnym rogu \textbf{IDE} kliknij w przycisk \texttt{Go Live} jak na obrazku poniżej - {rys. \ref{rys 6.1}}
\begin{figure}[H]
    \centering
    \includegraphics[width=\linewidth]{figures/6.1.eps}\\
    \caption{Przycisk \texttt{Go Live} rozszerzenia \texttt{Live Server}}\label{rys 6.1}
\end{figure}
    
\end{enumerate}
